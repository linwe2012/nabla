301 bool EvaluateMacros(std::string& line, size_t p, std::string& tok, const Set<std::string>& macros) {
302  bool eval = true;
303  while (p < line.size()) {
304  bool should_exist = true;
305  tok.clear();
306 
307  while (isspace(line[p]))
308  ++p;
309  if (line[p] == '!') {
310  should_exist = false;
311  ++p;
312  }
313  while (isalnum(line[p]) || line[p] == '_') {
314  tok += line[p];
315  ++p;
316  }
317 
318  bool exist = macros.count(tok);
319  eval = !(exist ^ should_exist); // both stands will give true
320 
321  while (isspace(line[p]))
322  ++p;
323 
324  if (p > line.size() - 2) break;
325 
326  if (line[p] == '|' && line[p + 1] == '|') {
327  // true || ?
328  if (eval) break;
329  }
330 
331  if (line[p] == '&' && line[p + 1] == '&') {
332  // false && ?
333  if (!eval) break;
334  }
335  p += 2;
336  }
337  return eval;
338 }

341 std::string PreprocessShader(std::istream& is,
342  const Set<std::string>& macros,
343  uint32_t version) {
344  std::string line;
345  std::string tok;
346  std::ostringstream oss;
347  oss << "#version " << std::to_string(version) << " core\n";
348 
349  size_t p;
350  int cnt = 0;
351  while (is.good()) {
352  std::getline(is, line);
353  bool eval = true;
354  int id = -1;
355  for (p = 0; p < line.size(); ++p) {
356  if (p > line.size() - 3) break;
357 
358  if (line[p] == '$' && line[p + 1] == 'I' && line[p + 2] == 'D') {
359  id = p;
360  }
361 
362  if (line[p] == '/' && line[p + 1] == '/') {
363  tok.clear();
364  p += 2;
365  while (p < line.size() && !isspace(line[p])) {
366  tok += line[p];
367  ++p;
368  }
369  while (isspace(line[p])) {
370  ++p;
371  }
372 
374  if (tok == "@") {
375  eval = EvaluateMacros(line, p, tok, macros);
376  }
377 
379  else if (tok == "#ID") {
380  tok.clear();
381  while (!isspace(line[p])) {
382  tok += line[p];
383  ++p;
384  }
385 
386  id = atoi(tok.c_str());
387  }
388  break;
389  }
390  if (line[p] == '/' && line[p + 1] == '/' && line[p + 2] == '@') {
391  p += 3;
392  eval = EvaluateMacros(line, p, tok, macros);
393 
394  break;
395  }
396  }
397 
398  if (eval) {
399  if (id >= 0) {
400  std::string idstr = std::to_string(cnt);
401  line[id ] = idstr.size() > 0 ? idstr[0] : ' ';
402  line[id + 1ull] = idstr.size() > 1 ? idstr[1] : ' ';
403  line[id + 2ull] = idstr.size() > 2 ? idstr[2] : ' ';
404  ++cnt;
405  }
406  oss << line << '\n';
407  }
408  }
409 
410 
411  return oss.str();
412 }

498  glBindVertexArray(VAO);
499

501  glBufferData(GL_ARRAY_BUFFER, data.size_by_bytes, data.ptr, GL_DYNAMIC_DRAW);

506  }
507 
508  size_t size = 0;
509  for (const auto& ly : layouts) {
510  glVertexAttribPointer(ly.position, ly.count_per_vertex,
511  ly.type, ly.normalized ? GL_TRUE : GL_FALSE,
512  ly.stride_by_bytes, (void*)static_cast<size_t>(ly.first_element_offset_by_bytes));
513  glEnableVertexAttribArray(ly.position);
514  size += GetMaterialSizeByType(ly.material_type);
515  }

523  gMeshBuffer.buffer[handle.index()].size_vbo = data.size_by_bytes;

527  }

529  return handle;

556

683  MaterialHandle skybox,
684  ShaderHandle compute_shader) {
685  auto& f = gDefaultCaptureFrameBuffer;
686  auto pre_filter_shader = OpenHandle(compute_shader);
687  auto pre_id = OpenHandle(prefilter);
688  pre_filter_shader.Use();
689  pre_filter_shader.SetInt("skybox", 0);
690  pre_filter_shader.SetMat4("projection", f.captureProjection);
691  glActiveTexture(GL_TEXTURE0);
692  glBindTexture(GL_TEXTURE_CUBE_MAP, OpenHandle(skybox));
693 
694  glBindFramebuffer(GL_FRAMEBUFFER, f.captureFBO);
695  unsigned int maxMipLevels = 5;
696  for (unsigned int mip = 0; mip < maxMipLevels; ++mip)
697  {
698  // reisze framebuffer according to mip-level size.
699  unsigned int mipWidth = 128 * std::pow(0.5, mip);
700  unsigned int mipHeight = 128 * std::pow(0.5, mip);
701  glBindRenderbuffer(GL_RENDERBUFFER, f.captureRBO);
702  glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH_COMPONENT24, mipWidth, mipHeight);
703  glViewport(0, 0, mipWidth, mipHeight);
704 
705  float roughness = (float)mip / (float)(maxMipLevels - 1);
706  pre_filter_shader.SetFloat("roughness", roughness);
707  for (unsigned int i = 0; i < 6; ++i)
708  {
709  pre_filter_shader.SetMat4("view", f.captureViews[i]);
710  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_CUBE_MAP_POSITIVE_X + i, pre_id, mip);
711 
712  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
713  RenderCube();
714  }
715  }
716  glBindFramebuffer(GL_FRAMEBUFFER, 0);
717 }
