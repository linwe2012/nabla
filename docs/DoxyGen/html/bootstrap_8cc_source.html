71 
72 #define MODEL_INFO_MAN(type, name) \

74  NA_LOG_ERROR("expect model field " #name " not null in %s", it->first.as<std::string>().c_str()); \
75  continue; \
76  } \
77  else { \
78  model. name = it->second[#name].as<type>(); \

84  } \
85  else { \
86  model. name = it->second[#name].as<type>(); \
87  }
88 
89  NA_ASSET_MODEL_INFO_LIST(MODEL_INFO_MAN, MODEL_INFO_NOT);
90 #undef MODEL_INFO_MAN
91 #undef MODEL_INFO_NOT
92 
93  std::string id;
94  if (it->second["id"].IsDefined()) {
95  id = it->second["id"].as < std::string >();

99  }
