82 static float Phillips(const OceanParameters& op, int n_prime, int m_prime) {
83  glm::vec2 k(
84  op.Kx(n_prime),
85  op.Kz(m_prime)
86  );
87  float k_length = k.length();
88  if (k_length < 0.000001) return 0.0; // spare some computation
89  float k_length2 = k_length * k_length;
90  float k_length4 = k_length2 * k_length2;
91  auto w = op.w;
92  auto g = op.g;
93 
94  float k_dot_w = glm::normalize(glm::dot(k, w));
95  float k_dot_w2 = k_dot_w * k_dot_w * k_dot_w * k_dot_w * k_dot_w * k_dot_w;
96 
97  float w_length = w.length(); // the speed of the wind
98  float L = w_length * w_length / g;
99  float L2 = L * L;
100 
101  float damping = 0.001; //< a factor, is it emperical?
102  float l2 = L2 * damping * damping;
103 
104  return
105  op.A
106  * exp(-1.0f / (k_length2 * L2)) / k_length4
107  * k_dot_w2
108  * exp(-k_length2 * l2); //< again the factor
109 }

117 static Complex GaussRandComplex() {
118  float x1, x2, w;
119  do {
120  x1 = 2.f * UniformRand() - 1.f;
121  x2 = 2.f * UniformRand() - 1.f;
122  w = x1 * x1 + x2 * x2;
123  } while (w >= 1.f);
124 
125  w = sqrt((-2.f * log(w)) / w);
126  return Complex(x1 * w, x2 * w);
127 }

196 static Complex H_Tilde(OceanParameters& op, float t, int n_prime, int m_prime) {
197  int index = m_prime * (op.N + 1) + n_prime;
198 
199  auto& data = *op.data;
200  auto& sdata = data.shader_data[index];
201  auto& cdata = data.precompute[index];
202 
203  Complex htilde0(sdata.htilde0.x, sdata.htilde0.y);
204  Complex htilde0_conj(cdata.htilde0_conj.x, cdata.htilde0_conj.y);
205 
206  //TODO: Precompute omega
207  float omega_t = PreComputeOmega(op, n_prime, m_prime) * t;
208 
209  float cos_omega = cos(omega_t);
210  float sin_omega = sin(omega_t);
211 
212  Complex c0(cos_omega, sin_omega);
213  Complex c1(cos_omega, -sin_omega);
214 
215  return htilde0 * c0 + htilde0_conj * c1;
