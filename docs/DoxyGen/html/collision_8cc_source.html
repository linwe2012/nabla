24 void CollisionSystem::Update(Clock& clock)
25 {
26  // compute collision
27  for (auto& rigid_itr1 : rigids_) {
28  for (auto& rigid_itr2 : rigids_) {
29  if (rigid_itr1.entity == rigid_itr2.entity) continue;
30  Transform* trans1 = render->GetTransformEdit(rigid_itr1.entity);
31  Transform* trans2 = render->GetTransformEdit(rigid_itr2.entity);
32  if (trans1 == nullptr || trans2 == nullptr) {
33  continue;
34  }
35  glm::mat4 mat1 = glm::mat4(1.0f);
36  glm::mat4 mat2 = glm::mat4(1.0f);
37  mat1 = glm::translate(mat1, trans1->position);
38  mat2 = glm::translate(mat2, trans2->position);
39  mat1 = glm::scale(mat1, trans1->scale);
40  mat2 = glm::scale(mat2, trans2->scale);
41 
42  // if collide
43  if (IsCollide(rigid_itr1.component, rigid_itr2.component, mat1, mat2)) {
44  ComputeCollison(rigid_itr1.component, rigid_itr2.component, mat1, mat2);
45  }
46  }
47  }
48 
49  // update rigids
50  for (auto& rigid_itr : rigids_) {
51  Transform* trans = render->GetTransformEdit(rigid_itr.entity);
52  if (trans == nullptr) {
53  continue;
54  }
55  RigidBody& rigid = rigid_itr.component;
56 
57  trans->position += rigid.velocity * clock.GetLastFrameDurationFloat();
58  }

88 bool CollisionSystem::IsCollide(RigidBody r1, RigidBody r2, const glm::mat4& mat1, const glm::mat4& mat2)
89 {
90  //transform to world coordinate
91  TransformAABB(r1._min, r1._max, mat1);
92  TransformAABB(r2._min, r2._max, mat2);
93 
94  float dt = 0.2;
95  r1._max += r1.velocity * dt;
96  r1._min += r1.velocity * dt;
97  r2._max += r2.velocity * dt;
98  r2._min += r2.velocity * dt;
99 
100  return !((r1._max.x < r2._max.x) || (r2._max.x < r1._min.x)
101  || (r1._max.y < r2._max.y) || (r2._max.y < r1._min.y)
102  || (r1._max.z < r2._max.z) || (r2._max.z < r1._min.z));

106 void CollisionSystem::ComputeCollison(RigidBody& r1, RigidBody& r2, const glm::mat4& mat1, const glm::mat4& mat2)
107 {
108  // get center in world coordinate
109  glm::vec3 min1 = r1._min;
110  glm::vec3 max1 = r1._max;
111  glm::vec3 min2 = r2._min;
112  glm::vec3 max2 = r2._max;
113  TransformAABB(min1, max1, mat1);
114  TransformAABB(min2, max2, mat2);
115 
116  // compute normal direction
117  glm::vec3 center1 = 0.5f * (max1 + min1);
118  glm::vec3 center2 = 0.5f * (max2 + min2);
119  glm::vec3 normal_dir = glm::normalize(center2 - center1);
120 
121  // normal velocity
122  glm::vec3 v1_x = normal_dir * glm::dot(r1.velocity, normal_dir);
123  glm::vec3 v2_x = normal_dir * glm::dot(r2.velocity, normal_dir);
124 
125  // tangential velocity
126  glm::vec3 v1_y = r1.velocity - v1_x;
127  glm::vec3 v2_y = r2.velocity - v2_x;
128 
129  // new normal velocity
130  glm::vec3 u1_x = ((2 * r2.mass) * v2_x + (r1.mass - r2.mass) * v1_x) / (r1.mass + r2.mass);
131  glm::vec3 u2_x = ((2 * r1.mass) * v1_x + (r2.mass - r1.mass) * v2_x) / (r1.mass + r2.mass);
132 
133  // updata new velocity
134  r1.velocity = u1_x + v1_y;
135  r2.velocity = u2_x + v2_y;
136 }

139 void CollisionSystem::TransformAABB(glm::vec3& min, glm::vec3& max, const glm::mat4& mat)
140 {
141  glm::vec3 v[8];
142  v[0] = glm::vec3(min.x, max.y, max.z);
143  v[1] = glm::vec3(min.x, min.y, max.z);
144  v[2] = glm::vec3(max.x, min.y, max.z);
145  v[3] = glm::vec3(max.x, max.y, max.z);
146  v[4] = glm::vec3(max.x, max.y, min.z);
147  v[5] = glm::vec3(max.x, min.y, min.z);
148  v[6] = glm::vec3(min.x, min.y, min.z);
149  v[7] = glm::vec3(min.x, max.y, min.z);
150 
151  //transform
152  for (int i = 0; i < 8; i++) {
153  glm::vec4 tmp(v[i], 1);
154  tmp = mat * tmp;
155  v[i] = glm::vec3(tmp.x, tmp.y, tmp.z);
156  }
157  //reset AABB
158  ResetAABB(min, max);
159  UpdateAABB(v, 8, min, max);
160 }
