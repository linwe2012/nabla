11 const Vector<Command>* RadixSort(Vector<Vector<Command>*> cmds_set ) {
12  static Vector<Command> a;
13  static Vector<Command> b;
14  a.clear();
15  b.clear();
16 
17  uint64_t max = 0;
18  int n_cmds = 0;
19 
20  for (auto cmds : cmds_set) {
21  for (auto cmd : *(cmds)) {
22  if (cmd.sortkey.sortkey > max) {
23  max = cmd.sortkey.sortkey;
24  }
25  a.push_back(cmd);
26  }
27  n_cmds += cmds->size();
28  }
29 
30  if (b.size() < n_cmds) {
31  // a.resize(n_cmds);
32  b.resize(n_cmds);
33  }
34  uint64_t bucket[16];
35  enum : uint64_t {
36  bitmask = 0xF
37  };
38  int bucket_max_index[16];
39  uint64_t lsd = 0;
40 
41  while ((max >> (lsd * 4)) > 0)
42  {
43  memset(bucket, 0, 16 * sizeof(int));
44  memset(bucket_max_index, 0, 16 * sizeof(int));
45 
46  uint64_t shift = lsd * 4;
47 
48  for (auto& c : a)
49  {
50  // p->sortkey.sortkey % pow(16, lsd + 1);
51  ++bucket[(c.sortkey.sortkey >> shift) & bitmask];
52  }
53 
54  for (int i = 1; i < 16; ++i) {
55  bucket[i] += bucket[i - 1];
56  }
57 
58  for (int i = n_cmds - 1; i >= 0; --i) {
59  b[--bucket[(a[i].sortkey.sortkey >> shift) & bitmask]] = a[i];
60  }
61 
62  a.swap(b);
63 
64  max /= 16;
65  ++lsd;
66  }
67 
68  return &a;
69 }
